65 em_status_t
66 em_init(const em_conf_t *conf)
67 {
68  em_status_t stat;
69  int ret;
70 
71  RETURN_ERROR_IF(!conf, EM_FATAL(EM_ERR_BAD_POINTER), EM_ESCOPE_INIT,
72  "Conf pointer NULL!");
73 
74  stat = early_log_init(conf->log.log_fn, conf->log.vlog_fn);
75  RETURN_ERROR_IF(stat != EM_OK, EM_FATAL(stat),
76  EM_ESCOPE_INIT, "User provided log funcs invalid!");
77 
78  /* Sanity check: em_shm should not be set yet */
79  RETURN_ERROR_IF(em_shm != NULL,
80  EM_FATAL(EM_ERR_BAD_STATE), EM_ESCOPE_INIT,
81  "EM shared memory ptr set - already initialized?");
82  /* Sanity check: either process- or thread-per-core, but not both */
83  RETURN_ERROR_IF(!(conf->process_per_core ^ conf->thread_per_core),
84  EM_FATAL(EM_ERR_BAD_ARG), EM_ESCOPE_INIT,
85  "Select EITHER process-per-core OR thread-per-core!");
86 
87  /*
88  * Reserve the EM shared memory once at start-up.
89  */
90  odp_shm_t shm = odp_shm_reserve("em_shm", sizeof(em_shm_t),
91  ODP_CACHE_LINE_SIZE,
92  ODP_SHM_SINGLE_VA);
93 
94  RETURN_ERROR_IF(shm == ODP_SHM_INVALID, EM_ERR_ALLOC_FAILED,
95  EM_ESCOPE_INIT, "Shared memory reservation failed!");
96 
97  em_shm = odp_shm_addr(shm);
98 
99  RETURN_ERROR_IF(em_shm == NULL, EM_ERR_NOT_FOUND, EM_ESCOPE_INIT,
100  "Shared memory ptr NULL!");
101 
102  memset(em_shm, 0, sizeof(em_shm_t));
103 
104  /* Store shm handle, can be used in em_term() to free the memory */
105  em_shm->this_shm = shm;
106 
107  /* Store the given EM configuration */
108  em_shm->conf = *conf;
109 
110  if (!EM_API_HOOKS_ENABLE) {
111  memset(&em_shm->conf.api_hooks, 0,
112  sizeof(em_shm->conf.api_hooks));
113  }
114 
115  env_spinlock_init(&em_shm->init.lock);
116 
117  /* Initialize the log & error handling */
118  log_init();
119  error_init();
120 
121  /* Initialize libconfig */
122  ret = em_libconfig_init_global(&em_shm->libconfig);
123  RETURN_ERROR_IF(ret != 0, EM_ERR_OPERATION_FAILED, EM_ESCOPE_INIT,
124  "libconfig initialization failed:%d", ret);
125 
126  /* Initialize the synchronous API locks */
127  env_spinlock_init(&em_shm->sync_api.lock_global);
128  env_spinlock_init(&em_shm->sync_api.lock_caller);
129 
130  /*
131  * Initialize the physical-core <-> EM-core mapping
132  *
133  * EM-core <-> ODP-thread id mappings cannot be set up yet,
134  * the ODP thread id is assigned only when that thread is initialized.
135  * Set this mapping in core_map_init_local()
136  */
137  stat = core_map_init(&em_shm->core_map, conf->core_count,
138  &conf->phys_mask);
139 
140  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
141  "core_map_init() failed:%" PRI_STAT "", stat);
142 
143  /*
144  * Check validity of core masks for input_poll_fn and output_drain_fn.
145  *
146  * Masks must be a subset of logical EM core mask. Zero mask means
147  * that input_poll_fn and output_drain_fn are run on all EM cores.
148  */
149  stat = input_poll_init(&em_shm->core_map.logic_mask, conf);
150  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
151  "input_poll_init() failed:%" PRI_STAT "", stat);
152  stat = output_drain_init(&em_shm->core_map.logic_mask, conf);
153  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
154  "output_drain_init() failed:%" PRI_STAT "", stat);
155 
156  /*
157  * Initialize Event State Verification (ESV), if enabled at compile time
158  */
159  if (EM_ESV_ENABLE) {
160  stat = esv_init();
161  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
162  "esv_init() failed:%" PRI_STAT "", stat);
163  }
164 
165  /* Initialize EM callbacks/hooks */
166  stat = hooks_init(&conf->api_hooks);
167  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
168  "hooks_init() failed:%" PRI_STAT "", stat);
169 
170  /*
171  * Initialize the EM buffer pools and create the EM_DEFAULT_POOL based
172  * on config.
173  */
174  stat = pool_init(&em_shm->mpool_tbl, &em_shm->mpool_pool,
175  &conf->default_pool_cfg);
176 
177  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
178  "pool_init() failed:%" PRI_STAT "", stat);
179 
180  stat = event_init();
181 
182  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
183  "event_init() failed:%" PRI_STAT "", stat);
184 
185  stat = event_group_init(&em_shm->event_group_tbl,
186  &em_shm->event_group_pool);
187 
188  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
189  "event_group_init() failed:%" PRI_STAT "", stat);
190 
191  stat = queue_init(&em_shm->queue_tbl, &em_shm->queue_pool,
192  &em_shm->queue_pool_static);
193 
194  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
195  "queue_init() failed:%" PRI_STAT "", stat);
196 
197  stat = queue_group_init(&em_shm->queue_group_tbl,
198  &em_shm->queue_group_pool);
199 
200  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
201  "queue_group_init() failed:%" PRI_STAT "", stat);
202 
203  stat = atomic_group_init(&em_shm->atomic_group_tbl,
204  &em_shm->atomic_group_pool);
205 
206  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
207  "atomic_group_init() failed:%" PRI_STAT "", stat);
208 
209  stat = eo_init(&em_shm->eo_tbl, &em_shm->eo_pool);
210 
211  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
212  "eo_init() failed:%" PRI_STAT "", stat);
213 
214  em_queue_group_t default_queue_group = default_queue_group_create();
215 
216  RETURN_ERROR_IF(default_queue_group != EM_QUEUE_GROUP_DEFAULT,
217  EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
218  "default_queue_group_create() failed!");
219 
220  /* timer add-on */
221  if (conf->event_timer) {
222  stat = timer_init(&em_shm->timers);
223  RETURN_ERROR_IF(stat != EM_OK,
224  EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
225  "timer_init() failed:%" PRI_STAT "",
226  stat);
227  }
228 
229  /* Initialize basic Event Chaining support */
230  stat = chaining_init(&em_shm->event_chaining);
231  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT,
232  "chaining_init() failed:%" PRI_STAT "", stat);
233 
234  return EM_OK;
235 }

237 em_status_t
238 em_init_core(void)
239 {
240  em_locm_t *const locm = &em_locm;
241  odp_shm_t shm;
242  em_shm_t *shm_addr;
243  em_status_t stat;
244  int init_count;
245 
246  /* Lookup the EM shared memory on each EM-core */
247  shm = odp_shm_lookup("em_shm");
248  RETURN_ERROR_IF(shm == ODP_SHM_INVALID,
249  EM_ERR_NOT_FOUND, EM_ESCOPE_INIT_CORE,
250  "Shared memory lookup failed!");
251 
252  shm_addr = odp_shm_addr(shm);
253  RETURN_ERROR_IF(shm_addr == NULL, EM_ERR_BAD_POINTER, EM_ESCOPE_INIT_CORE,
254  "Shared memory ptr NULL");
255 
256  if (shm_addr->conf.process_per_core && em_shm == NULL)
257  em_shm = shm_addr;
258 
259  RETURN_ERROR_IF(shm_addr != em_shm, EM_ERR_BAD_POINTER, EM_ESCOPE_INIT_CORE,
260  "Shared memory init fails: em_shm:%p != shm_addr:%p",
261  em_shm, shm_addr);
262 
263  /* Store the EM core id of this core, returned by em_core_id() */
264  locm->core_id = phys_to_logic_core_id(odp_cpu_id());
265 
266  /* Check if input_poll_fn should be executed on this core */
267  stat = input_poll_init_local(&locm->do_input_poll,
268  locm->core_id, &em_shm->conf);
269  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT_CORE,
270  "input_poll_init_local() failed:%" PRI_STAT "", stat);
271 
272  /* Check if output_drain_fn should be executed on this core */
273  stat = output_drain_init_local(&locm->do_output_drain,
274  locm->core_id, &em_shm->conf);
275  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT_CORE,
276  "output_drain_init_local() failed:%" PRI_STAT "", stat);
277 
278  /* Initialize core mappings not known yet in core_map_init() */
279  stat = core_map_init_local(&em_shm->core_map);
280  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT_CORE,
281  "core_map_init_local() failed:%" PRI_STAT "", stat);
282 
283  stat = queue_init_local();
284  RETURN_ERROR_IF(stat != EM_OK, stat, EM_ESCOPE_INIT_CORE,
285  "queue_init_local() failed:%" PRI_STAT "", stat);
286 
287  /*
288  * Initialize timer add-on. If global init was not done (config),
289  * this is just a NOP
290  */
291  stat = timer_init_local();
292  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_INIT_CORE,
293  "timer_init_local() failed:%" PRI_STAT "", stat);
294 
295  env_spinlock_lock(&em_shm->init.lock);
296  init_count = ++em_shm->init.em_init_core_cnt;
297  env_spinlock_unlock(&em_shm->init.lock);
298 
299  /* Now OK to call EM APIs */
300 
301  /* Print info about the Env&HW when the last core has initialized */
302  if (init_count == em_core_count()) {
303  em_queue_group_t def_qgrp;
304 
305  def_qgrp = default_queue_group_update();
306  RETURN_ERROR_IF(def_qgrp != EM_QUEUE_GROUP_DEFAULT,
307  EM_ERR_LIB_FAILED, EM_ESCOPE_INIT_CORE,
308  "default_queue_group_update() failed!");
309 
310  daemon_eo_create();
311  print_em_info();
312 
313  /* Last */
314  em_shm->init.em_init_done = 1;
315  }
316 
317  env_sync_mem();
318 
319  return EM_OK;
320 }

322 em_status_t
323 em_term(const em_conf_t *conf)
324 {
325  odp_event_t odp_ev_tbl[EM_SCHED_MULTI_MAX_BURST];
326  event_hdr_t *ev_hdr_tbl[EM_SCHED_MULTI_MAX_BURST];
327  em_event_t em_ev_tbl[EM_SCHED_MULTI_MAX_BURST];
328  odp_queue_t odp_queue;
329  em_status_t stat;
330  int num_events;
331  int ret;
332  bool esv_ena = esv_enabled();
333 
334  (void)conf;
335 
336  if (em_shm->conf.event_timer)
337  timer_term();
338 
339  /*
340  * Flush all events in the scheduler.
341  * Scheduler paused during return from em_dispatch()
342  */
343  odp_schedule_resume();
344  /* run loop twice: first with sched enabled and then paused */
345  for (int i = 0; i < 2; i++) {
346  do {
347  num_events =
348  odp_schedule_multi_no_wait(&odp_queue, odp_ev_tbl,
349  EM_SCHED_MULTI_MAX_BURST);
350  if (num_events <= 0)
351  break;
352  /*
353  * Events might originate from outside of EM and need init.
354  */
355  event_init_odp_multi(odp_ev_tbl, em_ev_tbl/*out*/,
356  ev_hdr_tbl/*out*/, num_events,
357  true/*is_extev*/);
358  if (esv_ena)
359  evstate_em2usr_multi(em_ev_tbl, ev_hdr_tbl,
360  num_events, EVSTATE__TERM);
361 
362  em_free_multi(em_ev_tbl, num_events);
363  } while (num_events > 0);
364 
365  odp_schedule_pause();
366  }
367 
368  stat = chaining_term(&em_shm->event_chaining);
369  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_TERM,
370  "chaining_term() failed:%" PRI_STAT "", stat);
371 
372  ret = em_libconfig_term_global(&em_shm->libconfig);
373  RETURN_ERROR_IF(ret != 0, EM_ERR_LIB_FAILED, EM_ESCOPE_TERM,
374  "EM config term failed:%d");
375 
376  stat = pool_term(&em_shm->mpool_tbl);
377  RETURN_ERROR_IF(stat != EM_OK, EM_ERR_LIB_FAILED, EM_ESCOPE_TERM,
378  "pool_term() failed:%" PRI_STAT "", stat);
379 
380  /*
381  * Free the EM shared memory
382  */
383  ret = odp_shm_free(em_shm->this_shm);
384  RETURN_ERROR_IF(ret != 0, EM_ERR_LIB_FAILED, EM_ESCOPE_TERM,
385  "odp_shm_free() failed:%d", ret);
386  /* Set em_shm = NULL to allow a new call to em_init() */
387  em_shm = NULL;
388 
389  return EM_OK;
390 }

392 em_status_t
393 em_term_core(void)
394 {
395  em_status_t stat = EM_OK;
396 
397  if (em_core_id() == 0)
398  daemon_eo_shutdown();
399 
400  /* Stop timer add-on. Just a NOP if timer was not enabled (config) */
401  stat |= timer_term_local();
402 
403  /* Delete the local queues */
404  stat |= queue_term_local();
405 
406  return stat == EM_OK ? EM_OK : EM_ERR;
407 }
