36 /**
37  * @file
38  * @defgroup em_hooks API-hooks
39  * Event Machine API-callback hooks.
40  * @{
41  *
42  * EM API-callback hook functions can be registered for a selected set of
43  * EM APIs. The EM APIs in question are mostly fast path APIs, like em_send(),
44  * em_alloc() and em_free(). Control APIs generally do not need hook support.
45  * A registered user provided hook function will be called by EM each time the
46  * corresponding API is called.
47  * API-callback hooks enables the user to gather statistics, trace program and
48  * event flow etc. API hooks should not change the state of the events etc.
49  * they receive as arguments, nor should they call the same API from within the
50  * hook to avoid hook recursion.
51  * Hook support is only available when EM_API_HOOKS_ENABLE != 0.
52  *
53  * Do not include this file from the application, event_machine.h will
54  * do it for you.
55  */

61 /**
62  * API-callback hook for em_alloc() and em_alloc_multi().
63  *
64  * The hook will only be called for successful allocs, passing also the newly
65  * allocated 'events' to the hook.
66  * The state and ownership of the events must not be changed by the hook, e.g.
67  * the events must not be freed or sent etc. Calling em_alloc/_multi() within
68  * the alloc hook leads to hook recursion and must be avoided.
69  *
70  * @note em_alloc(): hook is called with events[1] and num_act = num_req = 1.
71  * @note em_alloc_multi(): hook is called with events[num_act] and
72  * num_req >= num_act >= 1
73  *
74  * API-callback hook functions can be called concurrently from different cores.
75  *
76  * @param[in] events[] Array of newly allocated events: 'events[num_act]'.
77  * Don't change the state of the array or the events!
78  * @param num_act The actual number of events allocated and written into
79  * 'events[]' (num_act <= num_req). This is the return val
80  * of em_alloc_multi() if at least one event was allocated
81  * (the hook is not called if no events were allocated).
82  * @param num_req The requested number of events to allocate,
83  * from em_alloc/_multi('num')
84  * @param size Event size >0, from em_alloc/_multi('size')
85  * @param type Event type to allocate, from em_alloc/_multi('type')
86  * @param pool Event pool handle, from em_alloc/_multi('pool')
87  *
88  * @see em_alloc(), em_alloc_multi() and em_hooks_register_alloc()
89  */
90 typedef void (*em_api_hook_alloc_t)(const em_event_t events[/*num_act*/],
91  int num_act, int num_req, size_t size,
92  em_event_type_t type, em_pool_t pool);
93 
94 /**
95  * API-callback hook for em_free() and em_free_multi().
96  *
97  * The hook will be called before freeing the actual events, after verifying
98  * that the events given are valid, thus the hook does not 'see' if the actual
99  * free-operation succeeds or fails.
100  * The state and ownership of the events must not be changed by the hook, e.g.
101  * the events must not be freed or sent etc. Calling em_free/_multi() within the
102  * free hook leads to hook recursion and must be avoided.
103  *
104  * @note em_free(): hook is called with events[1] and num = 1.
105  * @note em_free_multi(): hook is called with events[num] and num >= 1
106  *
107  * API-callback hook functions can be called concurrently from different cores.
108  *
109  * @param[in] events[] Array of events to be freed: 'events[num]'
110  * Don't change the state of the array or the events!
111  * @param num The number of events in the array 'events[]'.
112  *
113  * @see em_free(), em_free_multi() and em_hooks_register_free()
114  */
115 typedef void (*em_api_hook_free_t)(const em_event_t events[], int num);
116

128  * Calling em_send...() within the send hook leads to hook recursion and
129  * must be avoided.

131  * API-callback hook functions can be called concurrently from different cores.
132  *
133  * @see
134  */
135 typedef void (*em_api_hook_send_t)(const em_event_t events[], int num,
136  em_queue_t queue,
137  em_event_group_t event_group);

139 /**
140  * API-callback hooks provided by the user at start-up (init)
141  *
142  * EM API functions will call an API hook if given by the user through this
143  * struct to em_init(). E.g. em_alloc() will call api_hooks->alloc(...) if
144  * api_hooks->alloc != NULL. Not all hooks need to be provided, use NULL for
145  * unsused hooks.
146  *
147  * @note Not all EM API funcs have associated hooks, only the most used
148  * functions (in the fast path) are included.
149  * Notice that extensive usage or heavy processing in the hooks might
150  * significantly impact performance since each API call (that has a hook)
151  * will execute the extra code in the user provided hook.
152  *
153  * @note Only used if EM_API_HOOKS_ENABLE != 0
154  */
155 typedef struct {
156  /**
157  * API callback hook for _all_ alloc-variants:
158  * em_alloc() and em_alloc_multi()
159  * Initialize to NULL if unused.
160  */
161  em_api_hook_alloc_t alloc_hook;
162 
163  /**
164  * API callback hook for all free-variants:
165  * em_free() and em_free_multi()
166  * Initialize to NULL if unused.
167  */
168  em_api_hook_free_t free_hook;
169 
170  /**
171  * API callback hook used for _all_ send-variants:
172  * em_send(), em_send_multi(), em_send_group() and em_send_group_multi()
173  * Initialize to NULL if unused.
174  */
175  em_api_hook_send_t send_hook;
176 } em_api_hooks_t;
