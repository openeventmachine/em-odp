211 static inline void
212 evstate_update(ev_hdr_state_t *const evstate, const uint16_t api_op,
213  const void *const ev_ptr)
214 {
215  const em_locm_t *const locm = &em_locm;
216  const uint32_t *const pl_u32 = ev_ptr;
217  const queue_elem_t *const q_elem = locm->current.q_elem;
218 
219  if (!q_elem) {
220  evstate->eo = EM_EO_UNDEF;
221  evstate->queue = EM_QUEUE_UNDEF;
222  } else {
223  evstate->eo = q_elem->eo;
224  evstate->queue = q_elem->queue;
225  }
226  evstate->api_op = api_op;
227  evstate->core = locm->core_id;
228  if (ev_ptr)
229  evstate->payload_first = *pl_u32;
230 }

233 evhdr_update_state(event_hdr_t *const ev_hdr, const uint16_t api_op)
234 {
235  const void *ev_ptr = NULL;

237  if (em_shm->opt.esv.store_first_u32)
238  ev_ptr = em_event_pointer(ev_hdr->event);
239 
240  evstate_update(&ev_hdr->state, api_op, ev_ptr);
241 }

252 static void
253 evstate_error(const evstate_cnt_t cnt, const evstate_cnt_t exp, evhdl_t evhdl,
254  const event_hdr_t *const ev_hdr, const uint16_t api_op)
255 {
256  const em_event_t event = event_hdr_to_event(ev_hdr);
257  const void *ev_ptr = NULL;
258 
259  const int16_t free_cnt = cnt.free_cnt - FREE_CNT_INIT;
260  const int16_t free_exp = exp.free_cnt - FREE_CNT_INIT;
261 
262  const int16_t send_cnt = cnt.send_cnt - SEND_CNT_INIT;
263  const int16_t send_exp = exp.send_cnt - SEND_CNT_INIT;
264 
265  uint16_t evgen_cnt = cnt.evgen;
266  const uint16_t evgen_hdl = evhdl.evgen;
267 
268  uint16_t prev_op = ev_hdr->state.api_op;
269 
270  if (unlikely(prev_op > EVSTATE__LAST))
271  prev_op = EVSTATE__UNDEF;
272 
273  const evstate_info_t *err_info = &evstate_info_tbl[api_op];
274  const evstate_info_t *prev_info = &evstate_info_tbl[prev_op];
275 
276  ev_hdr_state_t err_state = {0}; /* store current invalid/error state */
277  ev_hdr_state_t prev_state; /* store previous good known state */
278 
279  char curr_eoname[EM_EO_NAME_LEN] = "(noname)";
280  char prev_eoname[EM_EO_NAME_LEN] = "(noname)";
281  char curr_qname[EM_QUEUE_NAME_LEN] = "(noname)";
282  char prev_qname[EM_QUEUE_NAME_LEN] = "(noname)";
283  char curr_payload[sizeof("0x12345678 ")] = "(n/a)";
284  char prev_payload[sizeof("0x12345678 ")] = "(n/a)";
285 
286  const eo_elem_t *eo_elem;
287  const queue_elem_t *q_elem;
288 
289  /* Check event!=undef to avoid error in em_event_pointer() */
290  if (likely(event != EM_EVENT_UNDEF))
291  ev_ptr = em_event_pointer(event);
292  /* Store the new _invalid_ event-state info into a separate struct */
293  evstate_update(&err_state, api_op, ev_ptr);
294  /* Copy the previous valid event-state info from the event-header */
295  prev_state = ev_hdr->state;
296 
297  /*
298  * Print the first 32bits of the event payload on failure,
299  * the option 'esv.store_payload_first_u32' affects storing during valid
300  * state transitions.
301  */
302  if (ev_ptr) {
303  snprintf(curr_payload, sizeof(curr_payload),
304  "0x%08" PRIx32 "", err_state.payload_first);
305  curr_payload[sizeof(curr_payload) - 1] = '\0';
306  }
307  /*
308  * Print the first 32 bits of the event payload for the previous valid
309  * state transition, if enabled in the EM runtime config file:
310  * 'esv.store_payload_first_u32 = true', otherwise not even stored.
311  */
312  if (em_shm->opt.esv.store_first_u32) {
313  snprintf(prev_payload, sizeof(prev_payload),
314  "0x%08" PRIx32 "", prev_state.payload_first);
315  prev_payload[sizeof(prev_payload) - 1] = '\0';
316  }
317 
318  /* current EO-name: */
319  eo_elem = eo_elem_get(err_state.eo);
320  if (eo_elem != NULL)
321  eo_get_name(eo_elem, curr_eoname, sizeof(curr_eoname));
322  /* previous EO-name: */
323  eo_elem = eo_elem_get(prev_state.eo);
324  if (eo_elem != NULL)
325  eo_get_name(eo_elem, prev_eoname, sizeof(prev_eoname));
326  /* current queue-name: */
327  q_elem = queue_elem_get(err_state.queue);
328  if (q_elem != NULL)
329  queue_get_name(q_elem, curr_qname, sizeof(curr_qname));
330  /* previous queue-name: */
331  q_elem = queue_elem_get(prev_state.queue);
332  if (q_elem != NULL)
333  queue_get_name(q_elem, prev_qname, sizeof(prev_qname));
334 
335  INTERNAL_ERROR(EM_FATAL(EM_ERR_EVENT_STATE), err_info->escope,
336  EVSTATE_ERROR_FMT,
337  event, evgen_hdl, evgen_cnt,
338  free_cnt, free_exp, send_cnt, send_exp,
339  prev_info->str, prev_state.core,
340  prev_state.eo, prev_eoname, prev_state.queue, prev_qname,
341  prev_payload,
342  err_info->str, err_state.core,
343  err_state.eo, curr_eoname, err_state.queue, curr_qname,
344  curr_payload,
345  evhdl.event, evhdl.evptr);
346 }

348 static inline em_event_t
349 esv_evinit(const em_event_t event, event_hdr_t *const ev_hdr,
350  const evstate_cnt_t init_cnt, const uint16_t api_op)
351 {
352  evhdl_t evhdl = {.event = event};
353 
354  evhdl.evgen = EVGEN_INIT;
355  ev_hdr->event = evhdl.event;
356 
357  /* Set initial counters (atomic) */
358  __atomic_store_n(&ev_hdr->state_cnt.u64, init_cnt.u64,
359  __ATOMIC_RELAXED);
360  /* Set initial state information (non-atomic) */
361  evhdr_update_state(ev_hdr, api_op);
362 
363  return evhdl.event;
364 }

385 static inline em_event_t
386 esv_em2usr(const em_event_t event, event_hdr_t *const ev_hdr,
387  const evstate_cnt_t cnt, const evstate_cnt_t exp_cnt,
388  const uint16_t api_op, const bool is_revert)
389 {
390  evhdl_t evhdl = {.event = event};
391  evstate_cnt_t new_cnt;
392 
393  /* Update state-count and return value of all counters (atomic) */
394  if (unlikely(is_revert)) {
395  /* Revert previous em2usr counter update on failed operation */
396  new_cnt.u64 = __atomic_add_fetch(&ev_hdr->state_cnt.u64,
397  cnt.u64, __ATOMIC_RELAXED);
398  } else {
399  /* Normal em2usr counter update */
400  new_cnt.u64 = __atomic_sub_fetch(&ev_hdr->state_cnt.u64,
401  cnt.u64, __ATOMIC_RELAXED);
402  }
403  evhdl.evgen = new_cnt.evgen;
404  ev_hdr->event = evhdl.event;
405 
406  if (unlikely(new_cnt.free_send_cnt != exp_cnt.free_send_cnt)) {
407  /* report fatal event-state error, never return */
408  evstate_error(new_cnt, exp_cnt, evhdl, ev_hdr, api_op);
409  /* never reached */
410  }
411 
412  /*
413  * Valid state transition, update state (non-atomic)
414  */
415  evhdr_update_state(ev_hdr, api_op);
416 
417  return evhdl.event;
418 }

459 static inline void
460 esv_usr2em(const em_event_t event, event_hdr_t *const ev_hdr,
461  const evstate_cnt_t cnt, const evstate_cnt_t exp_cnt,
462  const uint16_t api_op, const bool is_revert)
463 {
464  evhdl_t evhdl = {.event = event};
465  evstate_cnt_t new_cnt;
466 
467  /* Update state-count and return value of all counters (atomic) */
468  if (unlikely(is_revert)) {
469  /* Revert previous usr2em counter update on failed operation */
470  new_cnt.u64 = __atomic_sub_fetch(&ev_hdr->state_cnt.u64,
471  cnt.u64, __ATOMIC_RELAXED);
472  } else {
473  /* Normal usr2em counter update */
474  new_cnt.u64 = __atomic_add_fetch(&ev_hdr->state_cnt.u64,
475  cnt.u64, __ATOMIC_RELAXED);
476 
477  if (unlikely(new_cnt.evgen == EVGEN_MAX)) {
478  /* Avoid .evgen counter wrap */
479  const evstate_cnt_t sub = {.evgen = EVGEN_MAX - EVGEN_INIT,
480  .free_cnt = 0, .send_cnt = 0};
481  __atomic_fetch_sub(&ev_hdr->state_cnt.u64, sub.u64,
482  __ATOMIC_RELAXED);
483  }
484 
485  new_cnt.evgen -= 1;
486  }
487 
488  if (unlikely(new_cnt.free_send_cnt != exp_cnt.free_send_cnt ||
489  evhdl.evgen != new_cnt.evgen)) {
490  /* report fatal event-state error, never return */
491  evstate_error(new_cnt, exp_cnt, evhdl, ev_hdr, api_op);
492  /* never reached */
493  }
494 
495  /*
496  * Valid state transition, update state (non-atomic)
497  */
498  evhdr_update_state(ev_hdr, api_op);

501 static inline void
502 esv_usr2em_multi(const em_event_t ev_tbl[],
503  event_hdr_t *const ev_hdr_tbl[], const int num,
504  const evstate_cnt_t cnt, const evstate_cnt_t exp_cnt,
505  const uint16_t api_op, const bool is_revert)
506 {
507  const evhdl_t *const evhdl_tbl = (const evhdl_t *)ev_tbl;
508  evstate_cnt_t new_cnt;
509 
510  for (int i = 0; i < num; i++) {
511  /* Update state-count and return value of all counters (atomic) */
512  if (unlikely(is_revert)) {
513  /* Revert usr2em counter update on failed operation */
514  new_cnt.u64 =
515  __atomic_sub_fetch(&ev_hdr_tbl[i]->state_cnt.u64,
516  cnt.u64, __ATOMIC_RELAXED);
517  } else {
518  /* Normal usr2em counter update */
519  new_cnt.u64 =
520  __atomic_add_fetch(&ev_hdr_tbl[i]->state_cnt.u64,
521  cnt.u64, __ATOMIC_RELAXED);
522 
523  if (unlikely(new_cnt.evgen == EVGEN_MAX)) {
524  /* Avoid .evgen counter wrap */
525  const evstate_cnt_t sub = {.evgen = EVGEN_MAX - EVGEN_INIT,
526  .free_cnt = 0, .send_cnt = 0};
527  __atomic_fetch_sub(&ev_hdr_tbl[i]->state_cnt.u64, sub.u64,
528  __ATOMIC_RELAXED);
529  }
530 
531  new_cnt.evgen -= 1;
532  }
533 
534  if (unlikely(new_cnt.free_send_cnt != exp_cnt.free_send_cnt ||
535  evhdl_tbl[i].evgen != new_cnt.evgen)) {
536  /* report fatal event-state error, never return */
537  evstate_error(new_cnt, exp_cnt, evhdl_tbl[i],
538  ev_hdr_tbl[i], api_op);
539  /* never reached */
540  }
541 
542  /*
543  * Valid state transition, update state (non-atomic)
544  */
545  evhdr_update_state(ev_hdr_tbl[i], api_op);
546  }
547 }
