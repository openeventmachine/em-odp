351 /**
352  * 'Periodical' draining of output from EM, if needed.
353  *
354  * User provided function - EM calls this, if not NULL, in the dispatch loop on
355  * each core - set via 'em_conf.output.output_drain_fn'
356  *
357  * Draining of output events/pkts: EM will every once in a while call this
358  * user provided function to ensure that low rate buffered output is eventually
359  * sent out. Not needed if your EM output queues (EM_QUEUE_TYPE_OUTPUT) always
360  * sends all events out. Useful in situations where output is buffered and sent
361  * out in bursts when enough output has been gathered - single events or low
362  * rate flows may, without this function, never be sent out (or too late) if the
363  * buffering threshold has not been reached.
364  *
365  * @return number of events successfully drained and sent for output
366  */
367 typedef int (*em_output_drain_func_t)(void);
368 
369 /**
370  * Output function, user provided callback for queues of type
371  * EM_QUEUE_TYPE_OUTPUT.
372  *
373  * This function will be called by em_send*() when sending to a queue of type
374  * EM_QUEUE_TYPE_OUTPUT and EM will take care of correct function calling order
375  * based on the scheduling context type.
376  * The function can use em_sched_context_type_current() if it needs information
377  * about e.g. ordering requirements set by the parent scheduled queue.
378  *
379  * @param events List of events to be sent out (ptr to array of events)
380  * @param num Number of events (positive integer)
381  * @param output_queue Output queue that the events were sent to (em_send*())
382  * @param output_fn_args Extra arguments to indicate e.g. ordering requirement
383  * of the source context.
384  *
385  * @return number of events successfully sent (equal to num if all successful)
386  */
387 typedef int (*em_output_func_t)(const em_event_t events[],
388  const unsigned int num,
389  const em_queue_t output_queue,
390  void *output_fn_args);
