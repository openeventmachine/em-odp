31 /**
32  * @file
33  * Compile time configuration for the event timer
34  * Can be overridden via makefile
35  *
36  */
37 #ifndef EM_TIMER_CONF_H_
38 #define EM_TIMER_CONF_H_
39 
40 /*
41  * Global configuration for the EM timer
42  */
43 
44 #ifndef EM_ODP_TIMER_RESOL_DEF_NS
45 /* Default timer resolution in ns (if none given by the user) */
46 #define EM_ODP_TIMER_RESOL_DEF_NS (1000ULL * 1000ULL * 1ULL) /* 1ms */
47 #endif
48 
49 #ifndef EM_ODP_MAX_TIMERS
50 /* Max number of supported EM timers (ODP timer pools) */
51 #define EM_ODP_MAX_TIMERS 16
52 #endif
53 
54 #ifndef EM_ODP_DEFAULT_TMOS
55 /* Default number of simultaneous timeouts per timer (handle pool size) */
56 #define EM_ODP_DEFAULT_TMOS 1000
57 #endif
58 
59 #ifndef EM_ODP_DEFAULT_RING_TMOS
60 /* Default per ring timer number of timeouts. Total comes from runtime config */
61 #define EM_ODP_DEFAULT_RING_TMOS 100
62 #endif
63 
64 #ifndef EM_TIMER_TMO_STATS
65 /* use 0 to exclude timeout statistics support */
66 #define EM_TIMER_TMO_STATS 1
67 #endif
68 
69 #ifndef EM_TIMER_ACK_TRIES
70 /* periodic ack() watchdog - how many times to try setting new timeout on late situation.
71  * Setting this to 1 effectively disables catch up, i.e. ack will fail if late. Setting this to 0
72  * will enable ~forever retry. 3 or bigger is recommended
73  */
74 #define EM_TIMER_ACK_TRIES 5
75 #endif
76 
77 #endif /* EM_TIMER_CONF_H_ */
